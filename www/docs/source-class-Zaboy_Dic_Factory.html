 94:     /**
 95:      * Get ( or make) all params for _constructor of class which $reflectionObject was taken
 96:      * 
 97:      * If it is Service, ($serviceName isn't null) information from service config
 98:      * will be used
 99:      * 
100:      * @param /ReflectionObject $reflectionObject
101:      * @param string|null $serviceName
102:      * @return array  {@see ReflectionClass::newInstanceArgs()}
103:      */
104:     protected function _getConstructParamsValues( $reflectionObject, $serviceName= null)
105:     {
106:         $constructParams = array();
107:          //Get params for $className::__construct
108:         $reflectionConstruct = $reflectionObject->getMethod('__construct');
109:         /* @var $reflectionConstruct /ReflectionMethod */
110:         $reflectionParams = $reflectionConstruct->getParameters();
111:         // $reflectionParams array of ReflectionParameter
112:         $options = $this->_resolveOptions($reflectionParams, $serviceName);
113:         if (isset($options)) {
114:             $constructParams[] = $options;
115:             array_shift($reflectionParams);
116:         }            
117:         foreach ($reflectionParams as $reflectionParam) {
118:             $constructParams[] = $this->_getConstructParamValue($reflectionParam, $serviceName);
119:         }
120:         return $constructParams;
121:     }

124:     /**
125:      * Get ( or make) one param for _constructor of class which $reflectionObject was taken
126:      * 
127:      * If it is Service, ($serviceName isn't null) inforvation from service cofig
128:      * will be used. Else classes which are specified in {@see __construct()} will used.
129:      * 
130:      * @param /ReflectionParameter $reflectionParam
131:      * @param string $serviceName
132:      * @return object  {@see ReflectionClass::newInstanceArgs()}
133:      */
134:     protected function _getConstructParamValue( $reflectionParam, $serviceName= null)    
135:     {
136:         if ($reflectionParam->isOptional()) {
137:             // optional params isn't loaded
138:             $instance = null;
139:         }else{
140:             $constructParamService = $this
141:                 ->_resolveServiceForConstructParam($reflectionParam, $serviceName);
142:             if (isset($constructParamService)) {
143:                $instance = $this->createService($constructParamService);
144:             }else{                    
145:                 $constructParamClassObject = $reflectionParam->getClass();
146:                 if (!isset($constructParamClassObject)) {
147:                     require_once 'Zaboy/Dic/Exception.php';
148:                     throw new Zaboy_Dic_Exception(
149:                         "Cann't resolve class for param  - " . $reflectionParam->getName());
150:                 }
151:                 $constructParamClass = $constructParamClassObject->getName();
152:                 $instance = $this->createObject($constructParamClass);
153:             }
154:         }
155:         return $instance;
156:     }

158:     /**

163:      * @return array|null  {@see ReflectionClass::newInstanceArgs()}
164:      */
165:     protected function _resolveOptions( $reflectionParams, $serviceName= null)
166:     {        
167:         //extract option (if exist) from array $reflectionParams
168:         $isOptionsInConstructParams = isset($reflectionParams[0]) 
169:             && $reflectionParams[0]->getName() === 'options';    
170:         if ($isOptionsInConstructParams) {

172:                 $optionsFromConfig = $this->_servicesConfigs->getServiceOptions($serviceName);
173:                 $options = $optionsFromConfig;
174:             }else{
175:                 $options = array();
176:             }
177:         }else{
178:             $options = null;
179:         }           
180:         return $options;
