 24:    /**
 25:     * Param $where can be null, numeric, string or array<br>
 26:     * <br>
 27:     * <b>null, numeric, string</b><br>
 28:     * If param $where is not array - it is for "Primary Key" ( See {@link getIdentifier()} )
 29:     * <code>
 30:     * fetch (), fetch (null) -->> select 
 31:     * fetch("5") -->> where 'id'='5'
 32:     * fetch(5) -->> where 'id'=5
 33:     * fetch('>5') -->> where <b>'id' = '>5'</b> //but what do you think? :-) Use $where = array( 'id'=>...
 34:     * </code>
 35:     * <b>array</b><br>
 36:     * If param $where is array, it may be:
 37:     * <ul>
 38:     * <li>associative array 'keyName'=>'condition' without numeric indexes</li>
 39:     * <li>simple array of associative arrays</li>
 40:     * <br> 
 41:     * </ul> 
 42:     * <b>combine conditions via AND</b><br>
 43:     * If param $where is associative array - it means "keyName" => "condition"
 44:     * <code>
 45:     * fetch(array("id" => "10", "fild" => "> 5") -->> where 'id'=10 AND "fild">5
 46:     * fetch (array("id" => "<>666", "fild" => "<> '666'") -->> where 'id'="<>666" AND "fild"<>6666 //sic!
 47:     * fetch (array("fild" => 'NOT NULL') -->> where 'fild' IS NOT NULL
 48:     * fetch (array("fild" => 'NULL') -->> where 'fild' IS NULL
 49:     * </code>
 50:     * <b>combine conditions via OR</b><br>
 51:     * If param $where is simple array of arrays (subarray) - it means   subarray1 OR subarray2 OR ...
 52:     * <code>
 53:     * fetch(
 54:     *     array(
 55:     *         array("id" => " > 0", "fild" => " = 'ok'")
 56:     *         array("id" => " < 10")
 57:     *     )
 58:     * )
 59:     * -->> where "id" > 0 AND "fild" = 'ok' OR "id" < 10
 60:     * </code>
 61:     * <b>Note</b><br>
 62:     * <code>
 63:     * array('id' => '10') === array('id' => ' = 10')
 64:     * array('id' => '<> NULL') -->> WHERE 'id' <> 'NULL' // you have to use array('id' => 'NOT NULL'), if you want WHERE 'id' NOT NULL
 65:     * </code>
 66:     * <br>
 67:     * <b>Param $options</b><br>
 68:     * Can has 3 keys
 69:     * <ul>
 70:     * <li>ORDER</li>
 71:     * <li>LIMIT</li>
 72:     * <li>OFFSET</li>
 73:     * <br> 
 74:     * <code>
 75:     * array(
 76:     * 'ORDER' => '+id',
 77:     * 'LIMIT' => 10 // if nuul - it s nolimit
 78:     * 'OFFSET' => 0 // if 'OFFSET' => 10 start will be from 11 ( if count from 1)
 79:     * )
 80:     * </code>
 81:     * 
 82:     * @param string|array {@link Avz_DataStore_Interface::fetch()}
 83:     * @param array it may contains ORDER LIMIT ect.
 84:     * @return array|object of array's ArrayAccess,Traversable,Countable
 85:     */
